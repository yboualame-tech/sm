Create New API Endpoint\n\n**File: `/src/app/api/grades/route.ts`**\n\n```typescript\nimport { NextRequest, NextResponse } from \"next/server\";\nimport prisma from \"../../../lib/prisma\";\nimport { requireAuth, requireRole } from \"../../../lib/auth-middleware\";\n\n/**\n * GET /api/grades\n * ADMIN: Get all grades\n * USER: Get only grades for their students\n */\nexport async function GET(request: NextRequest) {\n  try {\n    const user = await requireAuth(request);\n\n    if (user.role === \"ADMIN\") {\n      const grades = await prisma.grade.findMany({\n        include: { student: true },\n      });\n      return NextResponse.json(grades);\n    }\n\n    // USER: Only their students' grades\n    const userStudentIds = await prisma.student\n      .findMany({\n        where: { userId: user.id },\n        select: { id: true },\n      })\n      .then((s) => s.map((x) => x.id));\n\n    const grades = await prisma.grade.findMany({\n      where: { studentId: { in: userStudentIds } },\n      include: { student: true },\n    });\n    return NextResponse.json(grades);\n  } catch (error) {\n    if (error instanceof NextResponse) return error;\n    console.error(\"GET grades error:\", error);\n    return NextResponse.json(\n      { error: \"Failed to fetch grades\" },\n      { status: 500 }\n    );\n  }\n}\n\n/**\n * POST /api/grades\n * Create grade for a student\n */\nexport async function POST(request: NextRequest) {\n  try {\n    const user = await requireAuth(request);\n    const { studentId, subject, score, semester } = await request.json();\n\n    // Verify ownership (or admin)\n    if (user.role !== \"ADMIN\") {\n      const student = await prisma.student.findUnique({\n        where: { id: studentId },\n      });\n      if (!student || student.userId !== user.id) {\n        return NextResponse.json(\n          { error: \"Not authorized\" },\n          { status: 403 }\n        );\n      }\n    }\n\n    const grade = await prisma.grade.create({\n      data: { studentId, subject, score, semester },\n    });\n\n    return NextResponse.json(grade, { status: 201 });\n  } catch (error) {\n    if (error instanceof NextResponse) return error;\n    console.error(\"POST grade error:\", error);\n    return NextResponse.json(\n      { error: \"Failed to create grade\" },\n      { status: 500 }\n    );\n  }\n}\n```\n\n### 3.
